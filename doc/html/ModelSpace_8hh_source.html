41struct Orbit
42{
43 int n;
44 int l;
45 int j2;
46 int tz2;
47 double occ; // particle=0, hole=1
48 double occ_nat; // occupation in the natural orbital basis (if we use natural orbitals) particle=0, hole=1
49 int cvq; // core=0, valence=1, qspace=2
50 int index;
51
52 Orbit(){};
53 Orbit(int n, int l, int j2, int tz2, double occ, int cvq, int index)
54 : n(n), l(l), j2(j2), tz2(tz2),occ(occ),occ_nat(occ),cvq(cvq),index(index) {};
55
56 Orbit(const Orbit& orb) : n(orb.n), l(orb.l), j2(orb.j2), tz2(orb.tz2),occ(orb.occ),occ_nat(orb.occ_nat),cvq(orb.cvq),index(orb.index) {}
57 bool operator==( const Orbit& rhs ) const { return ( n==rhs.n and l==rhs.l and j2==rhs.j2 and tz2==rhs.tz2 ); };
58};

63struct Ket // | pq >
64{
65 // Fields
66 Orbit* op;
67 Orbit* oq;
68 size_t p;
69 size_t q;
70 int dpq;
71 int phase_prefactor;
72
73 // Methods
74 Ket(){};
75 Ket(Orbit& op_in, Orbit& oq_in) : op(&op_in), oq(&oq_in), p(op_in.index), q(oq_in.index), dpq( op_in.index==oq_in.index ? 1 : 0)
76 {
77 phase_prefactor = ((op->j2+oq->j2)/2 + 1) % 2==0 ? 1 : -1;
78 };
79
80 int Phase(int J) { return phase_prefactor * (J%2==0 ? 1 : -1); }; // |pqJ> = Phase * |qpJ> This includes the fermionic minus sign.
81 int delta_pq() const {return dpq;};
82
83};

87struct Ket3 // | p q r >
88{
89 // Fields
90 size_t p;
91 size_t q;
92 size_t r;
93 Orbit* op;
94 Orbit* oq;
95 Orbit* oR; // unfortunate that this convention collides with the "or" keyword.
96 int Jpq;
97
98 // Methods
99 Ket3(){};
100 Ket3(Orbit& op_in, Orbit& oq_in, Orbit& oR_in)
101 : p(op_in.index), q(oq_in.index), r(oR_in.index), op(&op_in), oq(&oq_in), oR(&oR_in), Jpq(0) {};
102
103 Ket3(Orbit& op_in, Orbit& oq_in, Orbit& oR_in, int jpq)
104 : p(op_in.index), q(oq_in.index), r(oR_in.index), op(&op_in), oq(&oq_in), oR(&oR_in), Jpq(jpq) {};
105
106 bool operator==(Ket3& rhs) { return (p==rhs.p and q==rhs.q and r==rhs.r and Jpq==rhs.Jpq);};
107 bool operator!=(Ket3& rhs) { return not( (*this)== rhs ) ;};
108};
